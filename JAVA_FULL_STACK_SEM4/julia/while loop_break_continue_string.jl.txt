x = 1          # Initialize a variable   # Program demonstrating while loop, break, and continue
while x <= 5                    # While loop with break and continue
    
    if x % 2 == 0                        # Check if x is even
        println("$x is even")
    else
        println("$x is odd")
    end
    x += 1                # Increment x          
    if x == 4                # Break if x reaches 4
        println("Breaking the loop")
        break
    end
end
output: 1 is odd
	2 is even
	3 is odd
	Breaking the loop
