arr = [1, 2, 3, 4, 5]                   # Define an array # Program demonstrating arrays and its manipulations
println("First element: ", arr[1])              # Accessing elements
println("Last element: ", arr[end])
println("Length of the array: ", length(arr))             # Length of the array
push!(arr, 6)               # Adding elements to the end of the array
println("Array after adding 6 at the end: ", arr)
pop!(arr)               # Removing the last element
println("Array after removing the last element: ", arr)
insert!(arr, 3, 10)                  # Inserting an element at a specific index
println("Array after inserting 10 at index 3: ", arr)
deleteat!(arr, 2)           # Removing an element at a specific index
println("Array after deleting element at index 2: ", arr)

	Last element: 5

	Array after inserting 10 at index 3: [1, 2, 10, 3, 4, 5]
	Array after deleting element at index 2: [1, 10, 3, 4, 5]
arr1 = [1, 2, 3]                 # Define two arrays # Program demonstrating array operations
arr2 = [4, 5, 6]
arr_concat = vcat(arr1, arr2)              # Concatenation

println("Element-wise multiplication: ", arr_product)
