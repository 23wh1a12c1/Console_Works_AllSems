function factorial(n)   # Program demonstrating recursion
    if n == 0
        return 1
    else
        return n * factorial(n - 1)
    end
end
n = 5        # Call the factorial function
println("Factorial of $n is ", factorial(n))
output:  Factorial of 5 is 120
